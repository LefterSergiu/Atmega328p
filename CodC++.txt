#include <avr/io.h> 
#define F_CPU 1000000 
#include <util/delay.h> 

int main(void)
{
    DDRB = 0b11111100;
    int x=0; 
    while(1)
    {
               if (bit_is_clear(PINB,1)) 
               {
                     if (bit_is_set(PINB,0)) 
                    {
                          x=1; 
                    }
               }
        
             if (x==1) 
            {
                 PORTB |=(1<<PINB2); 
                _delay_ms(220); 
                PORTB &=~(1<<PINB2); 
                x=0; 
            }
     }
}
